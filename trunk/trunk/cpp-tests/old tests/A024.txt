 int main() {

    // Assignment operators.
    
    int a ;
    float f ;
    
    a = 12 ;
    f = 3.141592 ;
    a = f ;
    f = 12 ;
    
    a *= 5 ;
    a *= 3.5 ;
    
    f *= 12.4 ;
    f *= 5 ;
    
    a += 3 ;
    f += 3 ;
    
    a -= 2 ;
    f -= 3.4 ;
    
    a = 9 ;
    a /= 2 ;
    f = 9 ;
    f /= 2 ;
    
    a = 17 ;
    a %= 3 ;
    
    a = 25 ;
    a <<= 3 ;
    a >>= 3 ;
    
    a &= 5 ;
    a |= 10 ;
    a ^= 7 ;
    
    f ^= 5 ;
}
